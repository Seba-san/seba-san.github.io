class BagConfig:
    def __init__(self,filepath='',frame_id='',topic_find=''):
        #base='/home/seba/Respaldo/Doctorado/investigacion/simulacion/ws/src/slam_benchmark/bags/'
        self.filepath=filepath
        self.frame_id=frame_id
        self.topic_find=topic_find

    def is_msg(self,topic,msg):
        if self.topic_find==topic:
            msgg=self.get_msg(msg)    
            frame=msgg.header.frame_id
            if self.frame_id==frame:
                return True
            else:
                return False


class BagTf(BagConfig):
    def __init__(self,filepath='',frame_id='',topic_find=''):
        BagConfig.__init__(self,filepath,frame_id,topic_find)


    def get_msg(self,msg):

        """
        Dependiendo del tipo de mensaje se acomoda para que pueda ser leido de
        la misma forma.
        """
        return msg.transforms[0]

    def get_T_R(self,msgg):
        """
        Obtiene T y R en formato np array del mensaje
        """
        fun=msgg.transform.translation
        T=np.array([fun.x,fun.y,fun.z])
        fun=msgg.transform.rotation
        R=np.array([fun.w,fun.x,fun.y,fun.z])
        return T,R

        """
        Dependiendo del tipo de mensaje se acomoda para que pueda ser leido de

def str2array(txt='[a a]'):
    if type(txt)!=str:
        return txt

    txt=txt[1:-1]
    #txt=txt.split()
    a=np.fromstring(txt,sep=' ')
    a=a-np.zeros([1,len(a)])# Is for convert to a matrix
    return a.T


[documentos]def euler_from_quaternion(R):
        """
        Convert a quaternion into euler angles (roll, pitch, yaw)
        roll is rotation around x in radians (counterclockwise)
        pitch is rotation around y in radians (counterclockwise)
        yaw is rotation around z in radians (counterclockwise)
        """
        x=R[0,0]
        y=R[1,0]
        z=R[2,0]
        w=R[3,0]
        t0 = +2.0 * (w * x + y * z)
        t1 = +1.0 - 2.0 * (x * x + y * y)
        roll_x = math.atan2(t0, t1)
     
        t2 = +2.0 * (w * y - z * x)
        t2 = +1.0 if t2 > +1.0 else t2
        t2 = -1.0 if t2 < -1.0 else t2
        pitch_y = math.asin(t2)
     
        t3 = +2.0 * (w * z + x * y)
        t4 = +1.0 - 2.0 * (y * y + z * z)
        yaw_z = math.atan2(t3, t4)
     
        return roll_x, pitch_y, yaw_z # in radians




def quaternion2euler(data,i):
    R=str2array(data.iloc[i])
    rpy=euler_from_quaternion(R)
    return np.array([[rpy[0],rpy[1],rpy[2]]]).T
