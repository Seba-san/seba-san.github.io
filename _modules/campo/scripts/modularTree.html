    implementadas
    """
    def __init__(self):
        self.parameters=TreeParameters()
        self.branch=BranchNode()
        self.trunk=TrunkNode()
        self.switch={
                'seed':self.setSeed,
                'amountLeaf':self.parameters.setLeafAmount,
                'name':self.setName,
                'splitAngle':self.branch.setSplitAngle,
                'filepath':self.setFilePath
                }


[documentos]    def set(self,argd={}):
        """
        A partir de un diccionario modifica todos los parametros del arbol de
        forma generica.
        """
        for name,argu in argd.items():
            fun=self.switch.get(name)
            fun(argu)

        return
